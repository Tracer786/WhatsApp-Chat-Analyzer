    {
        if((s[start] >= 'a' && s[start] <= 'z') || (s[start] >= 'A' && s[start] <= 'Z') || (s[start] >= '0' && s[start] <= '9'))
        {
            return 1;

        {
            end--;

        {
            return 0;

        {
            start++;
            end--;
        }
    }

int getMaximizedFunds(int k, int f, vector<int>& price, vector<int>& profit) {
    int n = price.size();
    vector<int> funds(k + 1, 0);

    for (int i = 1; i <= k; i++) {
        int maxProfit = funds[i - 1];
        for (int j = 0; j < n; j++) {
            if (i >= price[j]) {
                maxProfit = max(maxProfit, funds[i - price[j]] + profit[j]);
            }
        }
        funds[i] = maxProfit;
    }

    return f + funds[k];
}

{

    int dp[n + 1][sum + 1] ;
    dp[0][0] = 1;

    for (int i = 1; i <= sum; i++)
        dp[0][i] = 0;

    for (int i = 1; i <= n; i++)
        dp[i][0] = 1;

    for (int i = 1; i <= n; i++)
    {
        for (int j = 1; j <= sum; j++)
        {

            if (a[i - 1] > j)
                dp[i][j] = dp[i - 1][j];

            else
            {
                dp[i][j] = dp[i - 1][j] + dp[i - 1][j - a[i - 1]];
            }
        }
    }

    return (dp[n][sum] == 0 ? 999 : dp[n][sum] ) ;
}
